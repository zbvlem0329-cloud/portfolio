프로젝트 기획 배경

일과 휴식을 동시에 즐기려는 트렌드(워케이션)에 적합한 공간 탐색 시스템 구현

기존 숙박 앱은 오피스 인프라(인터넷, 업무 데스크 등) 정보가 부족하여 워케이션 목적에 부적합했습니다. 본 프로젝트는 사무용 편의시설 중심 필터링 검색과 함께, 판매자의 매출 원장 및 정산 자동화 시스템을 하나의 통합 플랫폼에 담았습니다.

3. 다차원 복합 필터 검색

QueryDSL을 활용한 다차원 동적 필터 쿼리

키워드, 지역 정보, 복합 편의시설(AND 조건), 기간 및 숙소 수용 인원을 한 번에 검증하여 최적의 오피스/스테이 매칭 목록을 동적으로 반환하는 dynamic query를 구축했습니다.
SpaceRepositoryImpl.java java 
@Override
public List<SpaceEntity> searchListForPublic(SpaceSearchReqDto dto) {
    QSpaceEntity space = QSpaceEntity.spaceEntity;
    QMemberEntity member = QMemberEntity.memberEntity;

    List<BooleanExpression> conditions = new ArrayList<>();
    conditions.add(space.delYn.eq("N"));
    conditions.add(space.visibleYn.eq("Y"));

    BooleanExpression keyword = keywordContains(space, dto.getKeyword());
    if (keyword != null) conditions.add(keyword);

    BooleanExpression area = areaEq(space, dto);
    if (area != null) conditions.add(area);

    // 편의시설 AND 필터
    BooleanExpression arcades = hasAllArcades(space, dto.getArcadeIds());
    if (arcades != null) conditions.add(arcades);

    // 날짜/인원 기준 이용 가능한 스테이 보유 여부
    BooleanExpression available = hasAvailableStay(space, dto.getStartDate(), dto.getEndDate(), dto.getCapacity());
    if (available != null) conditions.add(available);

    return queryFactory
            .selectFrom(space)
            .leftJoin(space.seller, member).fetchJoin()
            .where(conditions.toArray(new BooleanExpression[0]))
            .orderBy(space.createdAt.desc())
            .fetch();
}

@Transactional
    public void confirmPayment(PaymentConfirmReqDto dto, String username) {


        if (paymentRepository.existsByPaymentKey(dto.getPaymentKey())) {
            throw new PaymentConfirmException(HttpStatus.CONFLICT.value(),
                    "이미 처리된 결제입니다. 예약 내역을 확인해 주세요.");
        }

        ReservationEntity reservation = reservationRepository.findById(dto.getReservationId())
                .orElseThrow(() -> new RuntimeException("예약 없음 (ID: " + dto.getReservationId() + ")"));

        if (username == null || reservation.getMember() == null ||
                !username.equals(reservation.getMember().getUsername())) {
            throw new AccessDeniedException("Payment confirmation is allowed only for the reservation owner.");
        }

        try {
            RestTemplate restTemplate = new RestTemplate();


            restTemplate.getMessageConverters()
                    .add(0, new org.springframework.http.converter.StringHttpMessageConverter(java.nio.charset.StandardCharsets.UTF_8));

            String url = "https://api.tosspayments.com/v1/payments/confirm";

            HttpHeaders headers = new HttpHeaders();
            headers.setContentType(MediaType.APPLICATION_JSON);

            String encodedKey = Base64.getEncoder()
                    .encodeToString((secretKey + ":").getBytes(StandardCharsets.UTF_8));
            headers.set("Authorization", "Basic " + encodedKey);

            Map<String, Object> body = Map.of(
                    "paymentKey", dto.getPaymentKey(),
                    "orderId", dto.getOrderId(),
                    "amount", dto.getAmount()
            );

            ResponseEntity<String> response = restTemplate.exchange(
                    url, HttpMethod.POST, new HttpEntity<>(body, headers), String.class
            );

            // 이제 이 로그에서 한글이 깨지지 않고 "method":"간편결제", "provider":"카카오페이"로 정상 출력됩니다!
            log.info("Toss response: {}", response.getBody());

            if (!response.getStatusCode().is2xxSuccessful()) {
                throw new RuntimeException("토스 승인 API 실패");
            }

            JsonNode root = objectMapper.readTree(response.getBody());

            String rawMethod = root.path("method").asText();
            PaymentMethod method = parsePaymentMethod(rawMethod, root);

            //  [수정] 카드 정보 추출 시 방어 코드 및 한글 정제 로직 결합
            String cardCompany = null;
            String cardNumber = null;

            if (root.has("card") && !root.path("card").isNull()) {
                JsonNode cardNode = root.path("card");

                //  company 대신 확실하게 존재하는 issuerCode(발급사 코드)를 읽어옵니다.
                String issuerCode = cardNode.path("issuerCode").asText(); // 예: "91"

                //  코드를 기반으로 한글 이름을 매핑합니다.
                cardCompany = cleanCardCompanyName(issuerCode);
                cardNumber = cardNode.path("number").asText();
            }
            else if (root.has("easyPay") && !root.path("easyPay").isNull()) {
                JsonNode easyPayNode = root.path("easyPay");
                if (easyPayNode.has("provider")) {
                    cardCompany = easyPayNode.path("provider").asText();
                }
            }
            // 카카오페이/네이버페이 등 간편결제 시에도 카드사 컬럼에 결제 주체 명칭을 남기기 위한 백업
            else if (root.has("easyPay") && !root.path("easyPay").isNull()) {
                JsonNode easyPayNode = root.path("easyPay");
                if (easyPayNode.has("provider")) {
                    cardCompany = easyPayNode.path("provider").asText(); // "카카오페이" 등이 그대로 저장됨
                }
            }

            // 토스가 준 실제 승인 시간 파싱 (2026-05-21T14:35:09+09:00 형태 대응)
            LocalDateTime approvedAt = LocalDateTime.now();
            if (root.has("approvedAt")) {
                approvedAt = OffsetDateTime.parse(root.path("approvedAt").asText()).toLocalDateTime();
            }

            // 성공 내역 저장
            PaymentEntity payment = new PaymentEntity();
            payment.setReservation(reservation);
            payment.setOrderId(dto.getOrderId());
            payment.setPaymentKey(dto.getPaymentKey());
            payment.setAmount(dto.getAmount());
            payment.setPaymentMethod(method); // 동적 매핑 완료
            payment.setCardCompany(cardCompany);
            payment.setCardNumber(cardNumber);
            payment.setStatus(PaymentStatus.SUCCESS);
            payment.setApprovedAt(approvedAt);

            paymentRepository.save(payment);
            reservation.completePayment();

            //  [추가] 쿠폰 사용 처리 로직
            if (reservation.getCoupon() != null) {
                // 1. 해당 예약에 연결된 쿠폰이 맞는지 확인하고 MemberCouponEntity를 가져옵니다.
                // (ReservationEntity에 member와 coupon이 있으므로 가능)
                MemberCouponEntity memberCoupon = memberCouponRepository
                        .findByMemberAndCouponId(reservation.getMember(), reservation.getCoupon())
                        .orElseThrow(() -> new EntityNotFoundException("사용자의 쿠폰 내역을 찾을 수 없습니다."));

                // 2. 쿠폰 사용 처리 및 수량 감소
                memberCoupon.useCoupon();
                reservation.getCoupon().decrementQty();

                log.info("▶️ [쿠폰 사용 완료] 예약 ID: {}, 쿠폰 ID: {}", reservation.getId(), reservation.getCoupon().getId());
            }

            // 1. 매출 원장 즉시 자동 생성
            salesService.recordSales(payment);

            NotificationCreateReqDto notificationCreateReqDto = NotificationCreateReqDto.builder()
                    .memberId(reservation.getStay().getSpace().getSeller().getId()) //알림 받을 멤버id 번호 Long타입 변수로 처리하면 L안붙여도 됩니다!
                    .type(NotificationType.PAYMENT_SUCCESS) // 알림타입
                    .content("[" + reservation.getStay().getName() + "] 숙소 결제가 완료되어 예약 승인 요청 중입니다.") // [수정] 동적 문구 변경
                    .redirectUrl("/seller/reservations") // 알림 클릭햇을시 보내고 싶은 url(셀러 승인거절 페이지 만들기)
                    .referenceId(reservation.getId()) // 알림관련 식별번호 예약번호 or 상품번호 or 쿠폰번호 or 결제번호
                    .build();

            notificationService.createNotification(notificationCreateReqDto);

        } catch (RestClientResponseException e) {
            String providerBody = e.getResponseBodyAsString(StandardCharsets.UTF_8);
            log.error("Toss payment confirm failed. status={}, body={}",
                    e.getStatusCode().value(), providerBody, e);
            paymentFailureRecorder.record(reservation, dto, providerBody);
            throw new PaymentConfirmException(HttpStatus.BAD_GATEWAY.value(),
                    "Toss 결제 승인 API가 실패했습니다. 잠시 후 새 결제로 다시 시도해 주세요.",
                    providerBody);
        } catch (PaymentConfirmException e) {
            throw e;
        } catch (Exception e) {
            log.error("결제 승인 중 예외 발생 -> 실패 기록 작업을 별도 트랜잭션으로 진행", e);

            //  [핵심 수정] 자신의 프록시 객체나 빈을 직접 호출하는 대신,
            // 별도 트랜잭션 수단을 동반한 내부 로직으로 유도하거나 안전하게 저장 처리를 위임
            paymentFailureRecorder.record(reservation, dto, e.getMessage());

            throw new PaymentConfirmException(HttpStatus.INTERNAL_SERVER_ERROR.value(),
                    "결제 승인 후 서버 처리 중 오류가 발생했습니다: " + e.getMessage());
        }
    }

RefundService.java java 
@Transactional
public void refundReservation(RefundRequestDto dto, String username) {
    // 1. 예약 데이터 조회 및 유효성 검증
    ReservationEntity reservation = reservationRepository.findById(dto.getReservationId())
            .orElseThrow(() -> new EntityNotFoundException("예약 내역을 찾을 수 없습니다."));

    if (reservation.getStatus() == ReservationStatus.REFUND_COMPLETED) {
        throw new IllegalStateException("이미 환불 처리가 완료된 예약입니다.");
    }

    // 환불 제약 조건 체크 (2주전 100%, 1주전 50%, 이후 0%)
    int refundRate = calculateRefundRate(reservation);
    if (refundRate == 0) {
        throw new IllegalStateException("환불 가능 기간이 지나 환불이 불가능합니다.");
    }

    PaymentEntity payment = paymentRepository.findByReservation(reservation)
            .orElseThrow(() -> new EntityNotFoundException("결제 이력을 찾을 수 없습니다."));

    String paymentKey = payment.getPaymentKey();
    if (paymentKey == null || paymentKey.isBlank()) {
        throw new IllegalStateException("결제 고유 키가 누락되었습니다.");
    }

    // 비율에 따른 최종 환불 금액 계산
    long originalAmount = payment.getAmount();
    long finalRefundAmount = (originalAmount * refundRate) / 100;

    String url = "https://api.tosspayments.com/v1/payments/" + paymentKey + "/cancel";

    HttpHeaders headers = new HttpHeaders();
    headers.setContentType(MediaType.APPLICATION_JSON);
    String auth = tossSecretKey + ":";
    String encodedAuth = Base64.getEncoder().encodeToString(auth.getBytes(StandardCharsets.UTF_8));
    headers.set("Authorization", "Basic " + encodedAuth);

    TossCancelReqDto tossBody = TossCancelReqDto.builder()
            .cancelReason(dto.getReason().getDescription())
            .cancelAmount(finalRefundAmount)
            .build();

    HttpEntity<TossCancelReqDto> requestEntity = new HttpEntity<>(tossBody, headers);

    String transactionKey = "TX_" + payment.getOrderId() + "_" + System.currentTimeMillis();
    LocalDateTime canceledAt = LocalDateTime.now();

    try {
        ResponseEntity<Map> response = restTemplate.postForEntity(url, requestEntity, Map.class);

        if (response.getStatusCode() == HttpStatus.OK && response.getBody() != null) {
            Map<String, Object> responseBody = response.getBody();
            List<Map<String, Object>> cancels = (List<Map<String, Object>>) responseBody.get("cancels");

            if (cancels != null && !cancels.isEmpty()) {
                Map<String, Object> latestCancel = cancels.get(0);
                if (latestCancel.get("transactionKey") != null) {
                    transactionKey = (String) latestCancel.get("transactionKey");
                }
                if (latestCancel.get("canceledAt") != null) {
                    canceledAt = ZonedDateTime.parse((String) latestCancel.get("canceledAt")).toLocalDateTime();
                }
            }

            reservation.updateStatus(ReservationStatus.REFUND_COMPLETED);
            payment.cancel(finalRefundAmount, dto.getReason().getDescription());
            salesService.handleCancel(payment.getId(), finalRefundAmount);

            RefundEntity refund = RefundEntity.createRefund(reservation, transactionKey, finalRefundAmount, dto.getReason(), canceledAt);
            refundRepository.save(refund);
        } else {
            throw new IllegalStateException("토스 페이먼츠 응답 오류");
        }
    } catch (Exception e) {
        throw new IllegalStateException("외부 결제사 취소 실패: " + e.getMessage());
    }
}

import React, { useState, useEffect } from 'react';
import styled from 'styled-components';
import { getSellerReservations, approveReservation, rejectReservation } from '../../api/sellerApi';

export default function ReservationPage() {
  const [reservations, setReservations] = useState([]);
  const [activeTab, setActiveTab] = useState('ALL'); // ALL, PENDING, RESERVED, CANCELLED
  const [loading, setLoading] = useState(true);

  //  1. 판매자(호스트)의 모든 예약 목록 조회
  useEffect(() => {
    fetchReservations();
  }, []);

  const fetchReservations = async () => {
    try {
      setLoading(true);
      const data = await getSellerReservations();
      setReservations(data);
    } catch (error) {
      console.error('예약 내역 조회 실패:', error);
    } finally {
      setLoading(false);
    }
  };

  //  2. 예약 요청 승인 처리 (Accept)
  const handleApprove = async (reservationId) => {
    if (!window.confirm('이 예약을 수락하시겠습니까?')) return;
    try {
      await approveReservation(reservationId);
      alert('예약이 성공적으로 수락되었습니다.');
      fetchReservations(); // 상태 업데이트를 위한 재조회
    } catch (error) {
      alert('예약 수락 중 오류가 발생했습니다.');
    }
  };

  //  3. 예약 요청 반려 및 결제 취소 처리 (Reject)
  const handleReject = async (reservationId) => {
    if (!window.confirm('이 예약을 반려하시겠습니까? 반려 시 결제 금액이 자동 환불됩니다.')) return;
    try {
      await rejectReservation(reservationId);
      alert('예약이 반려되고 취소(환불) 처리가 완료되었습니다.');
      fetchReservations(); // 상태 업데이트를 위한 재조회
    } catch (error) {
      alert('예약 반려 중 오류가 발생했습니다.');
    }
  };

  // 탭 필터링 필터링된 목록 구하기
  const filteredList = reservations.filter(item => {
    if (activeTab === 'ALL') return true;
    if (activeTab === 'PENDING') return item.status === 'PAYMENT_COMPLETED';
    if (activeTab === 'RESERVED') return item.status === 'RESERVED';
    if (activeTab === 'CANCELLED') {
      return item.status === 'USER_CANCELLED' || item.status === 'SELLER_CANCELLED' || item.status === 'REFUND_COMPLETED';
    }
    return true;
  });

  return (
    <Container>
      <TitleZone>
        <h2>호스트 예약 승인 관리</h2>
        <p>신규 접수된 워케이션 예약을 승인하거나 반려(자동 환불)할 수 있습니다.</p>
      </TitleZone>

      <TabZone>
        {['ALL', 'PENDING', 'RESERVED', 'CANCELLED'].map((tab) => (
          <TabButton 
            key={tab} 
            active={activeTab === tab} 
            onClick={() => setActiveTab(tab)}
          >
            {tab === 'ALL' && '전체 예약'}
            {tab === 'PENDING' && '승인 대기'}
            {tab === 'RESERVED' && '예약 확정'}
            {tab === 'CANCELLED' && '취소/환불'}
          </TabButton>
        ))}
      </TabZone>

      {loading ? (
        <LoadingText>데이터를 불러오는 중입니다...</LoadingText>
      ) : filteredList.length === 0 ? (
        <EmptyText>조회된 예약 내역이 존재하지 않습니다.</EmptyText>
      ) : (
        <CardList>
          {filteredList.map((item) => (
            <ReservationCard key={item.id}>
              <CardHeader>
                <StayName>{item.stayName}</StayName>
                <StatusBadge status={item.status}>{item.statusLabel}</StatusBadge>
              </CardHeader>
              <CardBody>
                <InfoItem><strong>예약자:</strong> {item.primaryGuestName} ({item.primaryGuestEmail})</InfoItem>
                <InfoItem><strong>이용 기간:</strong> {item.checkinDate} ~ {item.checkoutDate}</InfoItem>
                <InfoItem><strong>이용 인원:</strong> 성인 {item.guestCount}명</InfoItem>
                <InfoItem><strong>결제 금액:</strong> ₩{item.totalPrice.toLocaleString()}</InfoItem>
              </CardBody>
              {item.status === 'PAYMENT_COMPLETED' && (
                <ActionButtons>
                  <ApproveBtn onClick={() => handleApprove(item.id)}>예약 수락</ApproveBtn>
                  <RejectBtn onClick={() => handleReject(item.id)}>예약 반려</RejectBtn>
                </ActionButtons>
              )}
            </ReservationCard>
          ))}
        </CardList>
      )}
    </Container>
  );
}

@Transactional
public void cancelReservationBySeller(Long id, String sellerUsername) {
    ReservationEntity reservation = reservationRepository.findById(id)
            .orElseThrow(() -> new EntityNotFoundException("예약 건을 찾을 수 없습니다."));

    if (!reservation.getStay().getSpace().getSeller().getUsername().equals(sellerUsername)) {
        throw new IllegalArgumentException("해당 예약에 대한 취소 권한이 없습니다.");
    }
    if (reservation.getStatus() == ReservationStatus.RESERVED) {
        throw new IllegalStateException("이미 승인 확정된 예약은 취소할 수 없습니다.");
    }
    
    // 1. 예약 거절 및 토스 결제 자동 환불 호출
    reservation.cancelBySeller();
    refundService.processFullRefundBySystem(reservation);

    // 2. 쿠폰 재활성화 (쿠폰 수량 및 사용여부 복구)
    MemberCouponEntity memberCoupon = memberCouponRepository
            .findByMemberAndCouponId(reservation.getMember(), reservation.getCoupon())
            .orElseThrow(() -> new EntityNotFoundException("사용자의 쿠폰 내역을 찾을 수 없습니다."));
            
    reservation.getCoupon().restoreQty();
    memberCoupon.restoreCoupon();

public Page<ReservationAdminListResDto> findAdminReservationList(
        PageRequest pageRequest,
        String username,
        String guestName,
        Long reservationId,
        String sellerUsername
) {
    // 1. 컨텐츠 데이터 조회
    List<ReservationEntity> targetList = qf
            .selectFrom(qReservation)
            .join(qReservation.member, qMember).fetchJoin()
            .join(qReservation.stay, qStay)
            .join(qStay.space, qSpace)
            .join(qSpace.seller, qSeller)
            .where(
                    usernameEq(username),
                    guestNameContains(guestName),
                    reservationIdEq(reservationId),
                    sellerUsernameEq(sellerUsername)
            )
            .orderBy(qReservation.id.desc())
            .offset(pageRequest.getOffset())
            .limit(pageRequest.getPageSize())
            .fetch();

    // 2. 전체 카운트 쿼리 조회
    Long total = qf
            .select(qReservation.count())
            .from(qReservation)
            .join(qReservation.stay, qStay)
            .join(qStay.space, qSpace)
            .join(qSpace.seller, qSeller)
            .where(
                    usernameEq(username),
                    guestNameContains(guestName),
                    reservationIdEq(reservationId),
                    sellerUsernameEq(sellerUsername)
            )
            .fetchOne();

    long totalCount = (total != null) ? total : 0L;

    // 3. DTO 변환 및 반환
    List<ReservationAdminListResDto> adminDtoList = targetList.stream()
            .map(ReservationAdminListResDto::from)
            .toList();

    return new PageImpl<>(adminDtoList, pageRequest, totalCount);

private ReservationDetailResDto bundleReservationDetail(ReservationEntity reservation) {
    // 1. 예약 연동 숙소 및 공간 추출
    StayEntity stayEntity = reservation.getStay();
    if (stayEntity == null) {
        throw new EntityNotFoundException("연결된 숙소 상품 정보가 없습니다.");
    }
    SpaceEntity spaceEntity = stayEntity.getSpace();

    // 2. 숙소 옵션 및 사진 정보 로드
    List<StayOptionEntity> optionEntities = stayOptionRepository.findByStay(stayEntity);
    List<StayPictureEntity> pictures = stayPictureRepository.findByStayOrderBySortOrder(stayEntity);

    List<StayOption> options = optionEntities.stream()
            .map(StayOptionEntity::getStayOption)
            .toList();

    StayResDto stayResDto = StayResDto.from(stayEntity, options, pictures);

    // 3. 썸네일을 숙소 사진의 첫 번째로 설정
    String stayThumbnail = (pictures != null && !pictures.isEmpty())
            ? s3Service.getFileUrl(pictures.get(0).getFilePath())
            : null;

    // 4. SpaceResDto 생성 (썸네일 인자 변경)
    SpaceResDto spaceResDto = SpaceResDto.from(spaceEntity, List.of(stayResDto), stayThumbnail);

    // 5. 결제 및 파일 정보 로드
    PaymentEntity paymentEntity = paymentRepository.findByReservation(reservation).orElse(null);
    List<ReserveFileEntity> reserveFiles = reserveFileRepository.findByReservationEntity(reservation);

    return ReservationDetailResDto.of(reservation, spaceResDto, stayResDto, paymentEntity, reserveFiles, s3Service);
}

프로젝트 완수 소감

프로젝트를 진행하며 얻은 기술적 배움과 인사이트

1. 외부 API 연동과 데이터 구조에 대한 이해 • 경험: 처음으로 외부 API를 프로젝트에 도입하며 시스템 구조 파악에 시행착오를 겪었습니다.
• 인사이트: 하지만 Request(요청 파라미터)와 Response(응답 데이터)의 규격을 명확히 분석하고 매핑하는 것이 API 활용의 핵심임을 깨달았고, 이후 새로운 외부 기술을 도입할 때 공식 문서를 분석하는 노하우를 얻었습니다.
2. 운영(SM) 관점에서의 단일 장애점(SPOF) 리스크 인지 • 경험: 초기에는 개발 효율성을 위해 토스페이먼츠 단일 결제망만을 연동했습니다.
• 인사이트: 프로젝트 완료 후 실제 서비스 운영 관점에서 접근해 보니, 해당 API 망에 에러가 발생할 경우 결제 전체가 마비되는 치명적인 위험(단일 장애점)이 존재함을 깨달았습니다. 또한 장애 원인이 내부 로직인지 외부 API인지 추적하기 어렵다는 점을 통해, 실무에서는 장애 대응과 유지보수를 위해 결제 시스템의 분산 및 다중화 설계가 필수적이라는 시각을 갖게 되었습니다.
3. 초기 기획의 중요성과 API 명세 협의 • 경험: 프론트엔드와 백엔드 역할을 분담하여 개발하던 중, 기획 변경으로 인해 양측이 요구하는 데이터 스펙이 빈번하게 달라지며 개발 일정이 지연되는 경험을 했습니다.
• 인사이트: 이를 통해 본격적인 개발 착수 전 화면에 렌더링 될 필수 데이터를 기획 단계에서 완벽하게 확정하는 꼼꼼한 초기 설계가 프로젝트 일정 관리에 직결된다는 것을 깊이 체감했습니다.
